def bubble_sort(a):
    arr = a.copy() 
    for i in range(len(arr)):
        for j in range(i, len(arr)):
            if arr[i] > arr[j]:
                buff = arr[j]
                arr[j] = arr[i]
                arr[i] = buff
    return arr
